create table line_detail(
id int NOT NULL AUTO_INCREMENT,
line_name varchar(255),
PRIMARY KEY (id)
)

create table tree_structure_detail(
id int NOT NULL AUTO_INCREMENT,
structure_name varchar(255),
PRIMARY KEY (id)
)



create table line_temp(
id int NOT NULL AUTO_INCREMENT,
line_id int,
tree_id int,
PRIMARY KEY (id),
FOREIGN KEY (line_id) REFERENCES line_detail(id),
FOREIGN KEY (tree_id) REFERENCES tree_structure_detail(id)
)

create table tag_detail(
id int NOT NULL AUTO_INCREMENT,
tag_name varchar(255),
PRIMARY KEY (id)
)

create table tag_table(
id int NOT NULL AUTO_INCREMENT,
head_id int,
tail_id int,
tree_id int,
PRIMARY KEY (id),
FOREIGN KEY (tree_id) REFERENCES tree_structure_detail(id),
FOREIGN KEY (head_id) REFERENCES tag_detail(id),
FOREIGN KEY (tail_id) REFERENCES tag_detail(id)
)

create table info_table(
id int NOT NULL AUTO_INCREMENT,
tag_id int,
career_progression text,
domain_roles text,
projects text,
responsibilites text,
PRIMARY KEY (id),
FOREIGN KEY (tag_id) REFERENCES tag_detail(id)
)
